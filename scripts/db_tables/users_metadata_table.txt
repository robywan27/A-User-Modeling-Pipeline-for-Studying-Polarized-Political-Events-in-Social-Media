CREATE TABLE users_metadata (
	userid bigint NOT NULL,
	name text, 
	screen_name text, 
	location text, 
	latitude double precision,
	longitude double precision,
	description text, 
	verified boolean, 
	followers_count integer, 
	friends_count integer, 
	listed_count integer, 
	favorites_count integer, 
	statuses_count integer, 
	created_at timestamp with time zone,
	time_zone text, 
	geo_enabled boolean,
	utc_offset integer,
	lang text,	
	profile_image_url character varying(910),
	url text,
	protected boolean,	
	userrole text,
	botaverage integer,
    botactual integer,
	gender character varying(20), 
	age integer,
	ethnicity text,
    politicalview character varying(50),
	total_tweets integer DEFAULT 0,
	home_city character varying(100),
	home_city_full text,
	home_country character varying(100),
	home_cc character varying(5),
	home_admin_level text,
	home_location character varying(90),
	CONSTRAINT users_metadata_pkey PRIMARY KEY (userid)
)
